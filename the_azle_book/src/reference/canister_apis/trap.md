```typescript
import { bool, Canister, ic, query, text } from 'azle/experimental';

export default Canister({
    // traps with a message, stopping execution and discarding all state within the call
    trap: query([text], bool, (message) => {
        ic.trap(message);

        return true;
    })
});
```
